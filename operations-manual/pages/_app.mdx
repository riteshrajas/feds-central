import { useEffect } from 'react';
import Head from 'next/head';
import Script from 'next/script';
import { Analytics } from "@vercel/analytics/next"
import { SpeedInsights } from "@vercel/speed-insights/next"

export default function MyApp({ Component, pageProps }) {
  useEffect(() => {
    const handleRouteChange = (url) => {
      window.gtag('config', 'G-N3TT9XBHV0', {
        page_path: url,
      });
    };

    window.addEventListener('routeChangeComplete', handleRouteChange);
  }, []);

  return (
    <>
      <Analytics/>
      <SpeedInsights />


      <Head>
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <meta name="google-site-verification" content="I7N03ZJUU7O6piTRRawvhXpkn8ueQbj_93K4i7j88vE" />
      </Head>
      {/* Google tag (gtag.js) */}
      <Script src="https://www.googletagmanager.com/gtag/js?id=G-6MLHZ8LFC0" strategy="afterInteractive" />
      <Script id="google-analytics" strategy="afterInteractive">
        {`
            window.dataLayer = window.dataLayer || [];
            function gtag(){dataLayer.push(arguments);}
            gtag('js', new Date());

            gtag('config', 'G-6MLHZ8LFC0');
        `}
      </Script>

      <Component {...pageProps} />


    </>
  );
}
